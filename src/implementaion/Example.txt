public class Implementation extends Implementor{
	
	/**
	 * Called when game starts
	 */
	public void main() {
		//Ignore
		super.main();
		
		//Write code here
		
		addButton(new Button1());
		addButton(new Button2());
		
		//Ignore
		begin();
	}
	public void buttonPressed(AbstractButton button){
		//Name of Button
		String buttonName = button.getName();
		if(buttonName.equals("Button1")){
			button.getButton().setText("Button 1 was Pressed");
		}
		else if(buttonName.equals("Button2")){
			button.getButton().setText("Button 2 was Pressed");
		}
	}
	public class Button1 extends AbstractButton{
		public void init(){
			setBounds(0,0,50,50);
			setText("Button1");
			setVisible(true);
		}
	}
	public class Button2 extends AbstractButton{
		public void init(){
			setBounds(100,100,50,50);
			setText("Button2");
			setVisible(true);
		}
	}
}
